int findMax(int arr[], int n)
{
    int max = -10000, index = 0;
    for(int i=0; i<n; ++i)
    {
        if(max < arr[i]) 
        {
            max = arr[i];
            index = i;
        }
    }
    return index;
}

int numof_simbit(int a, int b)
{
    int count = 0;
    for(int i=0; i<32; ++i)
    {
        if(((a>>i) & 1) == ((b>>i) & 1)) { ++count; }
    }
    return count;
}
